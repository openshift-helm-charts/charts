NEXT STEPS:

1. Check the deployment status:

   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name=wave-autoscale

2. Wait for all pods to be ready (this may take a few minutes):

   kubectl wait --for=condition=ready pod -l app.kubernetes.io/name=wave-autoscale -n {{ .Release.Namespace }} --timeout=300s

{{- if .Values.spec.agent.enabled }}
3. Verify Wave Autoscale agents are running on all nodes:

   kubectl get daemonset -n {{ .Release.Namespace }} wave-autoscale-agent
{{- end }}

4. Access the Wave Autoscale Web Console:

   {{- if eq .Values.serviceSpec.type "LoadBalancer" }}

   # Get the LoadBalancer IP or hostname:
   kubectl get svc wave-autoscale-svc -n {{ .Release.Namespace }}

   # Access the web console at:
   # http://<EXTERNAL-IP>:{{ .Values.serviceSpec.webConsole.port }}

   {{- else if eq .Values.serviceSpec.type "NodePort" }}

   # Get the NodePort:
   kubectl get svc wave-autoscale-svc -n {{ .Release.Namespace }}

   # Access the web console at:
   # http://<NODE-IP>:<NODE-PORT>

   {{- else }}

   # Port-forward to access the web console locally:
   kubectl port-forward -n {{ .Release.Namespace }} svc/wave-autoscale-svc {{ .Values.serviceSpec.webConsole.port }}:{{ .Values.serviceSpec.webConsole.port }}

   # Then visit: http://localhost:{{ .Values.serviceSpec.webConsole.port }}

   {{- end }}

5. Access the Wave Autoscale Core API:

   kubectl port-forward -n {{ .Release.Namespace }} svc/wave-autoscale-svc {{ .Values.serviceSpec.core.port }}:{{ .Values.serviceSpec.core.port }}

   # API available at: http://localhost:{{ .Values.serviceSpec.core.port }}
